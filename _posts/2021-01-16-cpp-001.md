### 2-Dimensional Random Walk

```CPP
#include<iostream>
#include<cstdlib>
#include<fstream>
#include<cmath>
#include<ctime>

using namespace std;


int Move(double a)
{
    int direction;
    if(a>=0.5)
    {
        direction = 1;
    }
    else
    {
        direction =-1;
    }
    
    return direction;
}

double RandomNumber()
{
    return (double)rand()/(double)RAND_MAX;
}

int main()
{
    srand(time(0));

    int t = 0;
    int steps = 100000;
    int x = 0;
    int y = 0;

    ofstream myfile("RandomWalk.dat");
    for(int i=0;i<steps;i++)
    {
        double direction = RandomNumber();
        double a = RandomNumber();
        double b = RandomNumber();

        if (direction >= 0.5)
        {
            x += Move(a);
        }
        else
        {
            y += Move(b);
        }

        myfile << i << " " << x << " " << y << endl;
    }

    return 0;
}
```
