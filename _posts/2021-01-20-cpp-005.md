### Two-dimensional Ising model with Metropolis-Hastings algorithm


```cpp
#include<iostream>
#include<fstream>
#include<cstdlib>
#include<cmath>
#include<ctime>
#include<random>

using namespace std;

const int N=50;
static int lattice[N][N];
const float J=1;
int totalM = 0;
int totalE = 0;
float T=0.5;
float Tmax=4.5;
int iteration=1000;

// Random number generator
random_device rd{};
mt19937 gen{rd()};

uniform_int_distribution<int> dis{0,N-1};

// Initialize the lattice where T=0
void InitializeLattice()
{
    for (int i=0;i<N;i++)
    {
        for (int j=0;j<N;j++)
        {
            lattice[i][j] = 1;
        }
    }
}

void GetTotalMagnetization()
{
    for (int i=0;i<N;i++)
    {
        for (int j=0;j<N;j++)
        {
            totalM += lattice[i][j];
        }
    }
}

void GetTotalEnergy()
{
    for (int i=0;i<N;i++)
    {
        for (int j=0;j<N;j++)
        {
            totalE += -J*lattice[i][j]*
                        (lattice[(i+1)%N][j] + lattice[(N+i-1)%N][j]+
                         lattice[i][(j+1)%N] + lattice[i][(N+j-1)%N]);
        }
    }
}

double deltaE(int i, int j)
{
    double dE_ij;
    dE_ij += 2*J*lattice[i][j]*
            (lattice[(i+1)%N][j] + lattice[(N+i-1)%N][j]+
            lattice[i][(j+1)%N] + lattice[i][(N+j-1)%N]);

    return dE_ij;
}

void PrintLattice()
{
    for(int i=0;i<N;i++)
    {
        for(int j=0;j<N;j++)
        {
            cout << lattice[i][j] << " ";
        }
        cout << endl;
    }
}

double RandomNumberGenerator()
{
    double x = (double)rand()/(double)RAND_MAX;
    return x;
}

int RandomSite()
{
    return dis(gen);
}

double MeanArray(int *arr)
{
    double sum=0;
    for(int i=0;i<sizeof(arr);i++)
    {
        sum += arr[i];
    }
    return sum/(sizeof(arr));
}

// Metropolis-Hastings algorithm
void step()
{
    for (int n=0;n<N;n++)
    {
        int i = RandomSite();
        int j = RandomSite();
        double dE = deltaE(i,j);
        double p = RandomNumberGenerator();

        if(dE<0)
        {
            lattice[i][j] *= -1;
            totalM += 2*lattice[i][j];
            totalE += dE;
        }
        else if(dE>0 && p<exp(-dE/T))
        {
            lattice[i][j] *= -1;
            totalM += 2*lattice[i][j];
            totalE += dE;
        }
    }
}

int main()
{
    clock_t begin=clock();
    
    // Initialization step
    InitializeLattice();
    GetTotalMagnetization();
    GetTotalEnergy();
    
    // Define output file
    ofstream myfile("Ising.dat");

    // Iteration
    while(T<Tmax)
    {
        int M[iteration];
        int E[iteration];
        double meanM;
        double meanE;
        for(int iter=0;iter<iteration;iter++)
        {
            step();
            M[iter] = totalM;
            E[iter] = totalE;
        }

        meanM = MeanArray(M);
        meanE = MeanArray(E);

        myfile << T << " " << meanM << " " << meanE << endl;
        cout << T << " " << meanM << " " << meanE << endl;
        T += 0.05;
    }
    clock_t end=clock();
    cout<< "Execution time: " << " " << (end-begin)/CLOCKS_PER_SEC<<endl;
    
    return 0;
}
```
